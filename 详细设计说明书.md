```typescript
// 随机推荐算法
function getRandomShop(shops: Shop[]): Shop {
  // 输入：店铺列表
  // 处理：从列表中随机选择一个店铺
  // 输出：随机选中的店铺
  
  if (shops.length === 0) {
    return null;
  }
  
  // 使用Math.random生成0到数组长度-1之间的随机索引
  const randomIndex = Math.floor(Math.random() * shops.length);
  return shops[randomIndex];
}
```

#### 2.2.3 筛选算法

```typescript
// 店铺筛选算法
function filterShops(
  shops: Shop[],
  filters: {
    area?: string;
    type?: string;
    keyword?: string;
  }
): Shop[] {
  // 输入：全部店铺列表，筛选条件
  // 处理：按条件过滤店铺
  // 输出：符合条件的店铺列表
  
  return shops.filter(shop => {
    // 区域筛选
    if (filters.area && filters.area !== '全部' && shop.area !== filters.area) {
      return false;
    }
    
    // 类型筛选
    if (filters.type && filters.type !== '全部' && shop.type !== filters.type) {
      return false;
    }
    
    // 关键词搜索（匹配店铺名或地址）
    if (filters.keyword) {
      const keyword = filters.keyword.toLowerCase();
      const matchName = shop.name.toLowerCase().includes(keyword);
      const matchAddress = shop.address?.toLowerCase().includes(keyword);
      if (!matchName && !matchAddress) {
        return false;
      }
    }
    
    return true;
  });
}
```

```typescript
// 收藏状态管理
interface FavoriteState {
  isFavorited: boolean;    // 是否已收藏
  favoriteId?: number;     // 收藏记录ID
}

// 切换收藏状态
async function toggleFavorite(
  userId: number,
  shopId?: number,
  dishId?: number
): Promise<FavoriteState> {
  // 检查当前收藏状态
  const currentState = await checkFavorite(userId, shopId, dishId);
  
  if (currentState.isFavorited) {
    // 已收藏，取消收藏
    await cancelFavorite(currentState.favoriteId);
    return { isFavorited: false };
  } else {
    // 未收藏，添加收藏
    const newFavoriteId = await addFavorite(userId, shopId, dishId);
    return { isFavorited: true, favoriteId: newFavoriteId };
  }
}
```

```typescript
// 计算店铺平均评分
function calculateAverageRating(reviews: Review[]): number {
  if (reviews.length === 0) {
    return 0;
  }
  
  const totalRating = reviews.reduce((sum, review) => sum + review.rating, 0);
  const averageRating = totalRating / reviews.length;
  
  // 四舍五入保留一位小数
  return Math.round(averageRating * 10) / 10;
}
```

```typescript
// 按距离排序
function sortByDistance(shops: Shop[]): Shop[] {
  return shops.sort((a, b) => {
    // 提取距离数值（去掉"m"后缀）
    const distanceA = parseFloat(a.distance.replace('m', ''));
    const distanceB = parseFloat(b.distance.replace('m', ''));
    
    return distanceA - distanceB;
  });
}
```

```typescript
// 关键词搜索匹配
function matchKeyword(text: string, keyword: string): boolean {
  if (!text || !keyword) {
    return false;
  }
  
  return text.toLowerCase().includes(keyword.toLowerCase());
}

// 多字段搜索
function multiFieldSearch<T extends Record<string, any>>(
  items: T[],
  keyword: string,
  fields: (keyof T)[]
): T[] {
  if (!keyword) {
    return items;
  }
  
  return items.filter(item => {
    return fields.some(field => {
      const value = item[field];
      if (typeof value === 'string') {
        return matchKeyword(value, keyword);
      }
      return false;
    });
  });
}
```
